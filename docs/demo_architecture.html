CREATE SEMANTIC VIEW SV_ENROLLMENTS
  TABLES (
    enrollments AS ANALYTICS.VW_ENROLLMENTS_BASE PRIMARY KEY enrollment_id
  )
  DIMENSIONS (
    enrollments.enrollment_status AS enrollment_status COMMENT = 'Active, Completed, or Dropped'
  )
  METRICS (
    enrollments.enrollment_count AS COUNT(*) COMMENT = 'Total number of enrollments',
    enrollments.avg_score AS AVG(final_score) COMMENT = 'Average final score'
  );
